## 第三步：设置 LD_LIBRARY_PATH

在 _ensure_playwright_browsers 函数最开头（诊断日志之后），加一段动态构建 LD_LIBRARY_PATH 的代码：

import subprocess, os

def _build_nix_ld_path():
    """从 Nix store 收集所有 /lib 路径注入 LD_LIBRARY_PATH"""
    nix_packages = [
        "atk", "at-spi2-atk", "at-spi2-core", "mesa", "libxkbcommon",
        "alsa-lib", "pango", "cairo", "gdk-pixbuf", "gtk+3", "nss",
        "nspr", "libXcomposite", "libXdamage", "libXrandr", "libXfixes",
        "libxcb", "libX11", "dbus", "expat", "cups", "libdrm", "glib", "libgbm"
    ]
    lib_paths = set()
    for root, dirs, files in os.walk("/nix/store"):
        if root.endswith("/lib") and any(pkg in root for pkg in nix_packages):
            lib_paths.add(root)
        # 不递归太深
        if root.count("/") > 6:
            dirs.clear()
    return ":".join(lib_paths)

# 只在 Playwright 下载的 chrome 场景生效（Nix .chrome-wrapped 不需要）
if not os.path.exists(nix_chrome_wrapped_path):  # 用已有的判断逻辑
    extra_ld = _build_nix_ld_path()
    current_ld = os.environ.get("LD_LIBRARY_PATH", "")
    os.environ["LD_LIBRARY_PATH"] = f"{extra_ld}:{current_ld}" if current_ld else extra_ld
    print(f"已注入 LD_LIBRARY_PATH，共 {len(extra_ld.split(':'))} 个路径")
